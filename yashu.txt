from difflib import get_close_matches
responses = {
    "hello": "Hello! How are you?"
    "what is your name": "My name is Hindi Bot.",
    "what can you do": "I can talk to you in Hindi.",
    "bye": "Goodbye! see you soon;"
}
def chatbot():
    print("Hindi Bot is running. type 'exit' to quit.\n")
    while True:
        user_input = input("you: ").strip().lower()
        if user_input == "exit":
            print("Bot: thank you,goodbye!")
            break
            match = get_close_matches(user_input, responses.keys(), n=1, cutoff=0.6)
            if match:
                print("Bot:",responses[match[0]])
            else:
                print("Bot: sorry,I didn't understandthat.")
                chatbot()
